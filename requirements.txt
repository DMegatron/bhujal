# Bhujal Simple Node.js Application Requirements
# Backend Dependencies for the Simple EJS-based application

# Core Framework
express@^4.18.2

# Database
mongoose@^7.5.0

# Authentication & Session Management
express-session@^1.17.3
bcryptjs@^2.4.3
connect-flash@^0.1.1

# Validation & Security
express-validator@^7.0.1
helmet@^7.0.0

# Template Engine
ejs@^3.1.9

# Utilities
method-override@^3.0.0
dotenv@^16.3.1

# Development Dependencies
nodemon@^3.0.1

# Installation Instructions:
# 1. Make sure you have Node.js (v16 or higher) installed
# 2. Run: npm install express@^4.18.2 mongoose@^7.5.0 express-session@^1.17.3 bcryptjs@^2.4.3 express-validator@^7.0.1 connect-flash@^0.1.1 method-override@^3.0.0 helmet@^7.0.0 ejs@^3.1.9 dotenv@^16.3.1
# 3. For development: npm install -D nodemon@^3.0.1
# 4. Or simply run: npm install (if package.json exists)

# Runtime Requirements:
# - Node.js v16.x or higher
# - MongoDB 4.4 or higher (local or cloud)
# - NPM v8.x or higher

# Environment Variables Required:
# - MONGODB_URI (MongoDB connection string)
# - SESSION_SECRET (for session encryption)
# - PORT (optional, defaults to 3000)
# - NODE_ENV (development/production)
