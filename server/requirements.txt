# Bhujal Express API Server Requirements
# Backend API Dependencies for the Express-based server

# Core Framework
express@^4.18.2

# Database
mongoose@^8.0.3

# Authentication & Security
bcryptjs@^2.4.3
jsonwebtoken@^9.0.2
cors@^2.8.5
helmet@^7.1.0
express-rate-limit@^7.1.5

# Validation
express-validator@^7.0.1

# File Upload
multer@^1.4.5-lts.1

# HTTP Client
axios@^1.6.2

# Environment Configuration
dotenv@^16.3.1

# Development Dependencies
nodemon@^3.0.2
jest@^29.7.0

# Installation Instructions:
# 1. Make sure you have Node.js (v16 or higher) installed
# 2. Navigate to the server directory: cd server
# 3. Run: npm install
# 4. For development: npm run dev
# 5. For production: npm start

# Runtime Requirements:
# - Node.js v16.x or higher
# - MongoDB 4.4 or higher (local or cloud)
# - NPM v8.x or higher

# Development Scripts:
# - npm start: Start production server
# - npm run dev: Start development server with nodemon
# - npm test: Run test suite with Jest

# Environment Variables Required:
# - MONGODB_URI (MongoDB connection string)
# - JWT_SECRET (JWT signing secret)
# - JWT_EXPIRE (JWT expiration time, e.g., 30d)
# - PORT (optional, defaults to 5000)
# - NODE_ENV (development/production)
# - WEATHER_API_KEY (OpenWeather API key)
# - CORS_ORIGIN (Allowed CORS origin, e.g., http://localhost:3000)
